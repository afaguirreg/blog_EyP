// Include the libraries we need
#include <LiquidCrystal_PCF8574.h>

LiquidCrystal_PCF8574 lcd(0x27); // set the LCD address to 0x27 for a 16 chars and 2 line display

float Valor_ADC_1;
float Valor_ADC_2;
float Valor_ADC_3;
float sensor_1;
float sensor_2;
float sensor_3;

void setup() {
  // put your setup code here, to run once:
  lcd.begin(20, 4); // initialize the lcd
  delay(200);
  lcd.setBacklight(255);
  lcd.print("Sensor 1:");
  lcd.setCursor(0,1);
  lcd.print("Sensor 2:");
  lcd.setCursor(0,2);
  lcd.print("Sensor 3:");
}

void loop() {
  // put your main code here, to run repeatedly:
  Valor_ADC_1 = analogRead(A0);
  Valor_ADC_2 = analogRead(A1);
  Valor_ADC_3 = analogRead(A2);

  sensor_1 = Valor_ADC_1 * 100 / 1023;
  sensor_2 = Valor_ADC_2 * 100 / 1023;
  sensor_3 = Valor_ADC_3 * 100 / 1023;

  lcd.setCursor(10,0);
  lcd.print(sensor_1,1);
  lcd.print("   ");
  lcd.setCursor(10,1);
  lcd.print(sensor_2,1);
  lcd.print("   ");
  lcd.setCursor(10,2);
  lcd.print(sensor_3,1);
  lcd.print("   ");
  delay(100);
}
