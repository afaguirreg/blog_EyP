#include <LiquidCrystal_PCF8574.h>

LiquidCrystal_PCF8574 lcd(0x27); // set the LCD address to 0x27 for a 16 chars and 2 line display

#define PIN_I0 16
#define PIN_I1 14
#define PIN_I2 12

#define I0 digitalRead(PIN_I0)
#define I1 digitalRead(PIN_I1)
#define I2 digitalRead(PIN_I2)

#define Q0 2
#define Q1 15
#define Q2 13

char texto[] = "I/O digitales";

byte alto[] = {0x00,0x1F,0x1F,0x1F,0x1F,0x1F,0x1F,0x00};
byte bajo[] = {0x00,0x1F,0x11,0x11,0x11,0x11,0x1F,0x00};

void estados() {
  lcd.setCursor(0,0);
  lcd.print(texto);
  lcd.setCursor(0,1);
  lcd.print("I: ");
  lcd.write(I0);
  lcd.write(I1);
  lcd.write(I2);
  lcd.setCursor(0,2);
  lcd.print("   012");
  lcd.setCursor(0,3);
  lcd.print("Q: ");
  lcd.write(digitalRead(Q0));
  lcd.write(digitalRead(Q1));
  lcd.write(digitalRead(Q2)); 
}

void setup() {
  // put your setup code here, to run once:
  pinMode(PIN_I0, INPUT);
  pinMode(PIN_I0, INPUT);
  pinMode(PIN_I2, INPUT);
  pinMode(Q0, OUTPUT);
  pinMode(Q1, OUTPUT);
  pinMode(Q2, OUTPUT);

  lcd.begin(20, 4); // initialize the lcd
  lcd.createChar(0, bajo);
  lcd.createChar(1, alto);
  delay(200);
  lcd.setBacklight(255);
}

void loop() {
  // put your main code here, to run repeatedly:
  estados();
  if (I0 && !I1)
    digitalWrite(Q0, HIGH);
  if (I1)
    digitalWrite(Q0, LOW);
}
