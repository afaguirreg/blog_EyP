// Include the libraries we need
#include <SPI.h>
#include <Wire.h>
#include <Adafruit_GFX.h>
#include <Adafruit_SSD1306.h>
#include <Arduino.h>
#include <SHT1x-ESP.h>

Adafruit_SSD1306 display = Adafruit_SSD1306(128, 32, &Wire);

// Specify data and clock connections and instantiate SHT1x object
#define dataPin  0
#define clockPin 5

// default to 5.0v boards, e.g. Arduino UNO
//SHT1x sht1x(dataPin, clockPin);

// if 3.3v board is used
SHT1x sht1x(dataPin, clockPin, SHT1x::Voltage::DC_3_3v);

char grados = 223;

void setup() {
  // put your setup code here, to run once:
  Serial.begin(9600);

  // SSD1306_SWITCHCAPVCC = generate display voltage from 3.3V internally
  display.begin(SSD1306_SWITCHCAPVCC, 0x3C); // Address 0x3C for 128x32
  display.setTextSize(1);
  display.setTextColor(SSD1306_WHITE);
  //Activar página de código 437
  display.cp437(true);
  display.display();
  delay(2000);
}

void loop() {
  // put your main code here, to run repeatedly:
  // Read values from the sensor
  float h = sht1x.readHumidity();
  float t = sht1x.readTemperatureC();

  display.clearDisplay();
  display.setCursor(0,0);
  display.print("Humedad:     ");
  display.print(h,1);
  display.println("%");
  display.print("Temperatura: ");
  display.print(t,1);
  display.print((char)grados);
  display.print("C");
  display.display();

  // Print the values to the serial port
  Serial.print("Humedad: ");
  Serial.print(h,1);
  Serial.print("% Temperatura: ");
  Serial.print(t,1);
  Serial.println("°C");
  delay(1000);
}
