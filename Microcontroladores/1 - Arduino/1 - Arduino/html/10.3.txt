#include <Servo.h>

Servo myservo;  // crea el objeto servo

int pos = 0;    // posici칩n del servo
int pin = 9;    // pin digital se침al control servo

void setup() {
  // put your setup code here, to run once:
   myservo.attach(pin);  // vincula el servo al pin digital
}

void loop() {
  // put your main code here, to run repeatedly:

  //varia la posici칩n de 0 a 180, con pausas de 15ms
  for (pos = 0; pos <= 180; pos += 1) 
  {
    myservo.write(pos);              
    delay(15);                       
  }

  //varia la posici칩n de 180 a 0, con pausas de 15ms
  for (pos = 180; pos >= 0; pos -= 1) 
  {
    myservo.write(pos);              
    delay(15);                       
  }
}
