#define boton_start_I0 16
#define boton_stop_I1 14
#define salida_Q0 9

#define estado_boton_start digitalRead(boton_start_I0)
#define estado_boton_stop digitalRead(boton_stop_I1)
#define salida_ON digitalWrite(salida_Q0, HIGH)
#define salida_OFF digitalWrite(salida_Q0, LOW)

void setup() {
  // put your setup code here, to run once:
  pinMode(boton_start, INPUT);
  pinMode(boton_stop, INPUT);
  pinMode(salida, OUTPUT);
}

void loop() {
  // put your main code here, to run repeatedly:
  if (estado_boton_start==1 && estado_boton_stop==0)
    salida_ON;
  if (estado_boton_stop==1)
    salida_OFF;
  delay(50);
}
