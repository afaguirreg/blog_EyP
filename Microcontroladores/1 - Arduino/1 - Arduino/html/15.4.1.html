void setup() {
  // put your setup code here, to run once:
  pinMode(boton_inc, INPUT);
  pinMode(10, OUTPUT);
  digitalWrite(10, HIGH);
  lcd.begin(16, 2);
  delay(200);
  lcd.print("Contador:");
}

void loop() {
  // put your main code here, to run repeatedly:
  if (estado_boton_inc)
    contador+=1;

  lcd.setCursor(0,1);
  lcd.print(contador);
  lcd.print("   ");
  delay(100);
}
