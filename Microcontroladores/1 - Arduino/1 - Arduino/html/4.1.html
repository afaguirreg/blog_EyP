#include <EEPROM.h>

 

#define boton_ON A0

#define boton_OFF A1

#define salida 8

 

#define estado_boton_ON digitalRead(boton_ON)

#define estado_boton_OFF digitalRead(boton_OFF)

#define salida_ON digitalWrite(salida, HIGH)

#define salida_OFF digitalWrite(salida, LOW)

 

void setup() {

  // put your setup code here, to run once:

  pinMode(boton_ON, INPUT);

  pinMode(boton_OFF, INPUT);

  pinMode(salida, OUTPUT);

  digitalWrite(salida, EEPROM.read(0));

}

 

void loop() {

  // put your main code here, to run repeatedly:

  if (estado_boton_ON)

  {

    salida_ON;

    EEPROM.write(0, 1);

  }

  if (estado_boton_OFF)

  {

    salida_OFF;

    EEPROM.write(0, 0);

  }

}
