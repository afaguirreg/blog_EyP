// include the library code:
#include <LiquidCrystal.h>
#include <ArduinoJson.h>

const int rs = 12, en = 11, d4 = 5, d5 = 4, d6 = 3, d7 = 2;
LiquidCrystal lcd(rs, en, d4, d5, d6, d7);

#define salida_LED 7

#define salida_LED_ON digitalWrite(salida_LED, HIGH)
#define salida_LED_OFF digitalWrite(salida_LED, LOW)

String Json_recibir = "";
boolean stringComplete;
bool LED;

void setup() {
  // put your setup code here, to run once:
  pinMode(salida_LED, OUTPUT);
  pinMode(10, OUTPUT);
  digitalWrite(10, HIGH);
  lcd.begin(16,2);
  delay(200);
  lcd.print("LED: ");
  Serial.begin(9600);
}

void loop() {
  // put your main code here, to run repeatedly:
  if (stringComplete==true)
  {
    StaticJsonDocument<300> Datos_recibir;
    DeserializationError err = deserializeJson(Datos_recibir, Json_recibir);

    if (err == DeserializationError::Ok) 
    {
      LED = Datos_recibir["LED"].as<bool>();
      switch (LED)
      {
        case true: salida_LED_ON; break;
        case false: salida_LED_OFF; break;
        default: break;
      }
    }

    lcd.setCursor(5,0);
    lcd.print(LED);

    // clear the string:
    Json_recibir = "";
    stringComplete = false;
  }
}

/*
  SerialEvent occurs whenever a new data comes in the hardware serial RX. This
  routine is run between each time loop() runs, so using delay inside loop can
  delay response. Multiple bytes of data may be available.
*/
void serialEvent() {
  Json_recibir = Serial.readStringUntil(10);
  stringComplete = true;
}
