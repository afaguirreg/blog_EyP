#include <ArduinoJson.h>

 

String Json_recibir = "";

bool stringComplete;

String numero;

 

int pins_unidades[7] = {8, 7, 6, 5, 4, 3, 2};

 

void write_digit(byte cifra) {

  for (byte i=0; i<7; i++)

  {

    digitalWrite(pins_unidades[i], (cifra>>i)&1);

  }

}

 

void setup() {

  // put your setup code here, to run once:

  Serial.begin(9600);

 

  for (byte i=0; i<7; i++)

  {

    pinMode(pins_unidades[i], OUTPUT);

    digitalWrite(pins_unidades[i], LOW);

  }

}

 

void loop() {

  // put your main code here, to run repeatedly:

  if (stringComplete==true)

  {

    // Allocate the JSON document

    // This one must be bigger than the sender's because it must store the strings

    StaticJsonDocument<300> Datos_recibir;

 

    // Read the JSON document from the "link" serial port

    DeserializationError err = deserializeJson(Datos_recibir, Json_recibir);

 

    if (err == DeserializationError::Ok)

    {

      // Print the values

      // (we must use as<T>() to resolve the ambiguity)

      numero = Datos_recibir["numero"].as<String>();

      switch (numero[0])

      {

        case '0': write_digit(0b00111111); break;

        case '1': write_digit(0b00000110); break;

        case '2': write_digit(0b01011011); break;

        case '3': write_digit(0b01001111); break;

        case '4': write_digit(0b01100110); break;

        case 53: write_digit(109); break;

        case 54: write_digit(124); break;

        case 55: write_digit(7); break;

        case 56: write_digit(127); break;

        case 57: write_digit(103); break;

        //Escribir c√≥digo para procesar A-F

        case 65: write_digit(119); break;

        case 98: write_digit(124); break;

        case 67: write_digit(57); break;

        case 100: write_digit(94); break;

        case 69: write_digit(121); break;

        case 70: write_digit(113); break;

        default: write_digit(0); break;

      }

    }

 

    // clear the string:

    Json_recibir = "";

    stringComplete = false;

  }

}

 

/*

  SerialEvent occurs whenever a new data comes in the hardware serial RX. This

  routine is run between each time loop() runs, so using delay inside loop can

  delay response. Multiple bytes of data may be available.

*/

void serialEvent() {

  Json_recibir = Serial.readStringUntil(10);

  stringComplete = true;

}
