// include the library code:
#include <LiquidCrystal.h>
#include <Servo.h>
#include <ArduinoJson.h>

const int rs = 12, en = 11, d4 = 5, d5 = 4, d6 = 3, d7 = 2;
LiquidCrystal lcd(rs, en, d4, d5, d6, d7);

#define boton_inc A1      //Boton incremento
#define boton_dec A2      //Boton decremento
#define boton_reset A3    //Boton reset

#define estado_boton_inc digitalRead(boton_inc)
#define estado_boton_dec digitalRead(boton_dec)
#define estado_boton_reset digitalRead(boton_reset)

#define salida_LED1 7
#define salida_LED2 8

#define salida_LED1_ON digitalWrite(salida_LED1, HIGH)
#define salida_LED1_OFF digitalWrite(salida_LED1, LOW)
#define salida_LED2_ON digitalWrite(salida_LED2, HIGH)
#define salida_LED2_OFF digitalWrite(salida_LED2, LOW)

boolean estado_boton_inc_ant;
boolean estado_boton_dec_ant;

#define pinPWM 6
#define pinServo 9

Servo myservo;  // crea el objeto servo

byte contador = 0;
float sensor;

String Json_recibir = "";
boolean stringComplete;
bool LED1, LED2;
int PWM = 0;
int angulo = 0;

void setup() {
  // put your setup code here, to run once:
  pinMode(boton_inc, INPUT);
  pinMode(boton_dec, INPUT);
  pinMode(boton_reset, INPUT);
  pinMode(salida_LED1, OUTPUT);
  pinMode(salida_LED2, OUTPUT);
  pinMode(10, OUTPUT);
  digitalWrite(10, HIGH);
  myservo.attach(pinServo);
  myservo.write(0);

  lcd.begin(20,4);
  delay(200);
  lcd.print("I:  D:  R:  C:");
  lcd.setCursor(0,1);
  lcd.print("Sensor: ");
  lcd.setCursor(0,2);
  lcd.print("LED1:   LED2:");
  lcd.setCursor(0,3);
  lcd.print("PWM:      Pos:");
  Serial.begin(9600);
}

void loop() {
  // put your main code here, to run repeatedly:
  if (estado_boton_inc!=estado_boton_inc_ant)
  {
    if (estado_boton_inc==1 && contador<100)
    {
      contador++;
    }
    estado_boton_inc_ant = estado_boton_inc;
    delay(5);
  }

  //Programar acci√≥n boton_dec

  if (estado_boton_reset==1)
  {
    contador = 0;
  }
  
  sensor = analogRead(A0);
  sensor *= 100.0 / 1023.0;

  lcd.setCursor(2,0);
  lcd.print(estado_boton_inc);

  //Programar visualizaci√≥n estado_boton_dec

  lcd.setCursor(10,0);
  lcd.print(estado_boton_reset);

  lcd.setCursor(14,0);
  lcd.print(contador);
  lcd.print("  ");

  lcd.setCursor(8,1);
  lcd.print(sensor,1);
  lcd.print("  ");

  StaticJsonDocument<300> Datos_enviar;
  Datos_enviar["estado_boton_inc"] = (estado_boton_inc ? "True" : "False");
  Datos_enviar["estado_boton_dec"] = (estado_boton_dec ? "True" : "False");
  Datos_enviar["estado_boton_reset"] = (estado_boton_reset ? "True" : "False");
  Datos_enviar["contador"] = contador;
  Datos_enviar["sensor"] = sensor;
  serializeJson(Datos_enviar, Serial);
  Serial.println();

  if (stringComplete==true)
  {
    StaticJsonDocument<300> Datos_recibir;
    DeserializationError err = deserializeJson(Datos_recibir, Json_recibir);

    if (err == DeserializationError::Ok) 
    {
      LED1 = Datos_recibir["LED1"].as<bool>();
      LED2 = Datos_recibir["LED2"].as<bool>();
      PWM = Datos_recibir["PWM"].as<int>();
      angulo = Datos_recibir["angulo"].as<int>();
    }

    switch (LED1)
    {
      case true: salida_LED1_ON; break;
      case false: salida_LED1_OFF; break;
      default: break;
    }

    switch (LED2)
    {
      case true: salida_LED2_ON; break;
      case false: salida_LED2_OFF; break;
      default: break;
    }

    lcd.setCursor(5,2);
    lcd.print(LED1);

    lcd.setCursor(13,2);
    lcd.print(LED2);

    lcd.setCursor(5,3);
    lcd.print("    ");
    lcd.setCursor(5,3);
    lcd.print(PWM);
    lcd.print("%");

    lcd.setCursor(15,3);
    lcd.print(angulo);
    lcd.print((char) 223);
    lcd.print("  ");

    analogWrite(pinPWM, PWM*255/100);

    myservo.write(angulo);

    // clear the string:
    Json_recibir = "";
    stringComplete = false;
  }
  delay(50);
}

/*
  SerialEvent occurs whenever a new data comes in the hardware serial RX. This
  routine is run between each time loop() runs, so using delay inside loop can
  delay response. Multiple bytes of data may be available.
*/
void serialEvent() {
  Json_recibir = Serial.readStringUntil(10);
  stringComplete = true;
}
