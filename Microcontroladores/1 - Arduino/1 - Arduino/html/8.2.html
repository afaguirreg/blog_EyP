#define boton_izquierda A1
#define boton_derecha A2
#define boton_stop A3

#define estado_boton_izquierda digitalRead(boton_izquierda)
#define estado_boton_derecha digitalRead(boton_derecha)
#define estado_boton_stop digitalRead(boton_stop)

byte estado_motor = 0; //1:izquierda 2:derecha
long sensor = 0;

void setup() {
  // put your setup code here, to run once:
  pinMode(boton_izquierda, INPUT);
  pinMode(boton_derecha, INPUT);
  pinMode(boton_stop, INPUT);
}

void loop() {
  // put your main code here, to run repeatedly:
  if (estado_boton_izquierda==1 && estado_boton_derecha==0 && estado_motor==0)
    estado_motor = 1;
  if (estado_boton_izquierda==0 && estado_boton_derecha==1 && estado_motor==0)
    estado_motor = 2;
  if (estado_boton_stop==1)
    estado_motor = 0;

  sensor = analogRead(A0);
  sensor = sensor * 255 / 1023;

  if (estado_motor==0)
  {
    analogWrite(9, 0);
    analogWrite(10, 0);
  }
  if (estado_motor==1)
  {
    analogWrite(9, sensor);
    analogWrite(10, 0);
  }
  if (estado_motor==2)
  {
    analogWrite(9, 0);
    analogWrite(10, sensor);
  }
  delay(50);
}
