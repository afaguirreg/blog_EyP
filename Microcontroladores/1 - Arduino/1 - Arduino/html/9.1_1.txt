// include the library code:
#include <LiquidCrystal.h>

const int rs = 12, en = 11, d4 = 5, d5 = 4, d6 = 3, d7 = 2;
LiquidCrystal lcd(rs, en, d4, d5, d6, d7);

#define boton_inc A0      //Boton incremento
#define boton_dec A1      //Boton decremento
#define boton_reset A2    //Boton reset

#define estado_boton_inc digitalRead(boton_inc)
#define estado_boton_dec digitalRead(boton_dec)
#define estado_boton_reset digitalRead(boton_reset)

byte contador = 0;
byte pBar = 0; // progress bar
byte i = 0;    // foor loop

byte pBrick[8] = {
  B11111,
  B11111,
  B11111,
  B11111,
  B11111,
  B11111,
  B11111,
  B11111,
};

void setup() {
  // put your setup code here, to run once:
  pinMode(boton_inc, INPUT);
  pinMode(boton_dec, INPUT);
  pinMode(boton_reset, INPUT);
  pinMode(10, OUTPUT);
  digitalWrite(10, HIGH);
  lcd.begin(16,2);
  lcd.createChar(0, pBrick);
  delay(200);
}

void loop() {
  // put your main code here, to run repeatedly:
  if (estado_boton_inc==1)
  {
    if (contador<255)
    {
      contador++;
    }
  }

  if (estado_boton_dec==1)
  {
    if (contador>0)
    {
      contador--;
    }
  }

  if (estado_boton_reset==1)
  {
    contador = 0;
  }

  lcd.setCursor(0,0);
  lcd.print("PWM: ");
  lcd.print(contador);
  lcd.print("   ");

  analogWrite(9, contador);

  pBar = map(contador, 0, 255, 0, 16);
  lcd.setCursor(0,1);
  for (i=1; i<=pBar; i++)
  {
    lcd.setCursor(i-1, 1);
    lcd.write(byte(0));
  }
  for (i=pBar; i<=16; i++)
  {
    lcd.setCursor(i, 1);
    lcd.print(" ");
  }
  delay(50);
}
