// include the library code:
#include <LiquidCrystal.h>

const int rs = 12, en = 11, d4 = 5, d5 = 4, d6 = 3, d7 = 2;
LiquidCrystal lcd(rs, en, d4, d5, d6, d7);

int sensor;
byte pBar = 0; // progress bar
byte i = 0;    // foor loop

byte pBrick[8] = {
  B11111,
  B11111,
  B11111,
  B11111,
  B11111,
  B11111,
  B11111,
  B11111,
};

void setup() {
  // put your setup code here, to run once:
  pinMode(10, OUTPUT);
  digitalWrite(10, HIGH);
  lcd.begin(16,2);
  delay(200);
  lcd.createChar(0, pBrick);
}

void loop() {
  // put your main code here, to run repeatedly:
  sensor = analogRead(A0);
  sensor = map(sensor,0,1023,0,255);

  lcd.setCursor(0,0);
  lcd.print("PWM: ");
  lcd.print(sensor);
  lcd.print("   ");

  analogWrite(9, sensor);

  pBar = map(sensor, 0, 255, 0, 16);
  lcd.setCursor(0,1);
  for (i=1; i<=pBar; i++)
  {
    lcd.setCursor(i-1, 1);
    lcd.write(byte(0));
  }
  for (i=pBar; i<=16; i++)
  {
    lcd.setCursor(i, 1);
    lcd.print(" ");
  }
  delay(50);
}
